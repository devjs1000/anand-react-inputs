## Install

```bash
npm install --save anand-react-inputs
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'anand-react-inputs'
import 'anand-react-inputs/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
